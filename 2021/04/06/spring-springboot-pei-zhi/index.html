spring_springboot配置

发表于2021-04-06|更新于2021-07-16
|阅读量:

张忘川专用配置！

yml基本配置（待补充）

application.yml

server:
  port: 8888
  servlet:
    context-path: /shiro


spring:
  application:
    name: shiro

  mvc:
    view:
      prefix: /
      suffix: .jsp
    static-path-pattern: /res/**

  web:
    resources:
      static-locations: [classpath:/haha/]
      
      
  datasource:
    type: com.alibaba.druid.pool.DruidDataSource
    driver-class-name: com.mysql.cj.jdbc.Driver
    username: root
    password: zxh228..
    url: jdbc:mysql://localhost:3306/shiro

  redis:
    port: 6379
    host: localhost
    database: 0


logging:
  level:
    com.zxh.shirotest.mapper : debug

JavaConfig（待补充）

继承WebMvcConfigurer

@Configuration
public class MyWebConfig implements WebMvcConfigurer {

    @Override
    public void addResourceHandlers(ResourceHandlerRegistry registry) {
        registry.addResourceHandler("/static/**").addResourceLocations("file:/usr/local/xuhui/");
    }
}
