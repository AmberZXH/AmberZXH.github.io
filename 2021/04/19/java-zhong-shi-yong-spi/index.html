/**
 * @author: AmberZxh

Main.java

package com.zxh;

import com.zxh.service.Pay;

import java.util.Iterator;
import java.util.ServiceLoader;

/**
 * @author: AmberZxh
 * @DateTime: 2021/4/19 10:19 下午
 * @description:
 */
public class Main {
    public static void main(String[] args) {
        ServiceLoader<Pay> load = ServiceLoader.load(Pay.class);
        for (Pay next : load) {
            next.model();
        }
    }
}
